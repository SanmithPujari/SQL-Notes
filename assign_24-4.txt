select * from books;

CREATE VIEW BookIssued AS
SELECT B.BOOK_ID,BOOK_NAME,QUANTITY_ISSUED 
FROM BOOKS B LEFT JOIN ISSUED I ON 
B.BOOK_ID=I.BOOK_ID

SELECT * FROM BOOKISSUED;

CREATE VIEW Author AS
SELECT B.BOOK_ID,AUTHOR_NAME
FROM BOOKS B;

SELECT * FROM BOOKISSUED;
SELECT * FROM AUTHOR;

SELECT B.BOOK_ID,BOOK_NAME,AUTHOR_NAME,QUANTITY_ISSUED
FROM BOOKISSUED B,AUTHOR A WHERE
B.BOOK_ID=A.BOOK_ID

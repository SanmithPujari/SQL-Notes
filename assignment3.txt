CREATE TABLE TEACHER(
TID INT,
TNAME VARCHAR(20),
SUBJECT VARCHAR(20),
PHNO INT,
PRIMARY KEY(TID) 
);

CREATE TABLE BRANCH(
BID INT,
BNAME VARCHAR(20),
PRIMARY KEY(BID)
);

create table student(
ROLL INT,
SNAME VARCHAR(20),
PHNO INT,
TID INT,
BRANCHID INT,
PRIMARY KEY(ROLL),
FOREIGN KEY(TID) REFERENCES TEACHER(TID),
FOREIGN KEY(BRANCHID) REFERENCES BRANCH(BID)
);
